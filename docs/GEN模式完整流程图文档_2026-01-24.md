```typescript
protected async doExecute(context: ExecutionContext): Promise<HandlerResult> {
  // ✅ 传递 sessionId 和 messageId 到 options（后端需要这些信息来保存附件）
  const genOptions = {
    ...context.options,
    frontend_session_id: context.sessionId,   // ✅ 会话 ID
    sessionId: context.sessionId,              // ✅ 向后兼容
    message_id: context.modelMessageId         // ✅ 消息 ID（关键！）
  };
  
  // 调用 llmService
  const results = await llmService.generateImage(
    context.text, 
    context.attachments,
    genOptions
  );

  // ✅ 后端已处理图片（返回 attachmentId, uploadStatus, taskId）
  const displayAttachments: Attachment[] = results.map((res: ImageGenerationResult) => ({
    id: res.attachmentId || uuidv4(),
    mimeType: res.mimeType || 'image/png',
    name: res.filename || `generated-${Date.now()}.png`,
    url: res.url,                              // 显示 URL（Base64 或 HTTP）
    uploadStatus: res.uploadStatus || 'pending',
    uploadTaskId: res.taskId
  }));

  return {
    content: `Generated images for: "${context.text}"`,
    attachments: displayAttachments,
    uploadTask: async () => ({ dbAttachments: displayAttachments })
  };
}
```

#### Step 5: UnifiedProviderClient.executeMode

**文件**: `frontend/services/providers/UnifiedProviderClient.ts`

```typescript
async executeMode(
  mode: string,           // 'image-gen'
  modelId: string,        // 'imagen-3.0-generate-002'
  prompt: string,         // 用户提示词
  attachments: [],        // 参考图片（可选）
  options: {              // 生成选项
    numberOfImages: 4,
    imageAspectRatio: '1:1',
    sessionId: '...',
    message_id: '...'     // ✅ 关键参数
  }
): Promise<any> {
  const requestBody = {
    modelId,
    prompt,
    attachments,
    options: { ...options }
  };
  
  // 统一路由: POST /api/modes/google/image-gen
  const url = `/api/modes/${this.id}/${mode}`;
  const response = await fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`
    },
    body: JSON.stringify(requestBody)
  });
  
  const data = await response.json();
  // 响应格式: { success: true, data: { images: [...] } }
  
  // 转换为 ImageGenerationResult[]
  return data.data.images.map((img: any) => ({
    url: img.url,
    mimeType: img.mimeType,
    filename: img.filename,
    attachmentId: img.attachmentId,
    uploadStatus: img.uploadStatus,
    taskId: img.taskId
  }));
}
```

#### AttachmentService.process_ai_result 详细流程

**文件**: `backend/app/services/common/attachment_service.py`

```python
async def process_ai_result(
    self,
    ai_url: str,          # AI 返回的 URL（Base64 或 HTTP）
    mime_type: str,
    session_id: str,
    message_id: str,
    user_id: str,
    prefix: str = 'generated'
) -> Dict[str, Any]:
    """
    处理 AI 返回的图片 URL
    
    流程:
    1. 判断 URL 类型（Base64 Data URL 或 HTTP URL）
    2. 创建 MessageAttachment 附件记录（写入数据库）
    3. 直接返回原始 URL 供前端显示
    4. 提交 Worker Pool 任务（异步上传到云存储）
    5. 返回附件信息
    """
    
    # Step 1: 判断 URL 类型
    url_type = "Base64" if ai_url.startswith('data:') else "HTTP"
    
    # Step 2: 生成附件 ID 和文件名
    attachment_id = str(uuid.uuid4())
    filename = f"{prefix}-{uuid.uuid4()}.png"
    
    # Step 3: 设置显示 URL（直接返回原始 URL）
    display_url = ai_url
    
    # Step 4: 创建数据库附件记录
    attachment = MessageAttachment(
        id=attachment_id,
        message_id=message_id,   # ✅ 关联到消息
        user_id=user_id,
        session_id=session_id,
        name=filename,
        mime_type=mime_type,
        temp_url=ai_url,         # 保存原始 URL（Base64）
        url='',                  # 云 URL（待上传完成后更新）
        upload_status='pending'
    )
    self.db.add(attachment)
    self.db.commit()
    
    # Step 5: 提交 Worker Pool 任务（Redis 队列）
    task_id = await self._submit_upload_task(
        session_id=session_id,
        message_id=message_id,
        attachment_id=attachment_id,
        source_ai_url=ai_url,    # 可以是 Base64 或 HTTP
        filename=filename,
        mime_type=mime_type
    )
    
    return {
        'attachment_id': attachment_id,
        'display_url': display_url,   # 前端立即显示
        'cloud_url': '',              # 待上传完成
        'status': 'pending',
        'task_id': task_id
    }
```

## 七、总结

GEN 模式的完整流程确保了：

1. **统一认证**: JWT Token → `require_current_user`
2. **统一凭证管理**: `get_provider_credentials` 解密 API Key
3. **统一路由**: `/api/modes/{provider}/{mode}`
4. **统一服务调用**: `ProviderFactory.create` → `GoogleService`
5. **统一结果处理**: `AttachmentService.process_ai_result`
6. **统一数据库持久化**: `MessageAttachment` + `UploadTask`
7. **统一异步上传**: Redis Queue → Worker Pool → 云存储

Virtual Try-On 需要接入 Step 7（AttachmentService）并确保前端传递 `sessionId`/`message_id` 才能实现完整的数据库持久化。
